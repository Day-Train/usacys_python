#8. Using the file school_prompt.txt, if the character ‘p’ is in a word, then add the word to a list called p_words.

with open('school_prompt.txt') as fp:
    p_words = []
    for line in fp:
        linel = line.split()
        for item in linel:
            if 'p' in item:
                p_words.append(item)
            
    print(p_words)

###

p_words = []
with open('school_prompt.txt') as fp:
    for line in fp:
        for word in line.split():
            if 'p' in word:
                p_words.append(word)
